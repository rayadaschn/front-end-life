function quickSort<T>(arr: Array<T>): T[] {
  if (arr.length < 2) return arr

  const left: Array<T> = []
  const right: Array<T> = []
  const mid = arr.splice(Math.floor(arr.length / 2), 1)[0]

  arr.forEach((item) => {
    if (item < mid) {
      left.push(item)
    } else {
      right.push(item)
    }
  })

  return [...quickSort(left), mid, ...quickSort(right)]
}
